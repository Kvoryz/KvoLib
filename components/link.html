HTML 
<a href="#" class="kvo-link-nav active">Home</a>
<a href="#" class="kvo-link-nav">Products</a>
CSS 
.kvo-link-nav {
  color: rgba(255, 255, 255, 0.6);
  text-decoration: none;
  padding: 8px 0;
  position: relative;
}

.kvo-link-nav::after {
  content: '';
  position: absolute;
  bottom: 0;
  left: 0;
  width: 100%;
  height: 2px;
  background: #667eea;
  transform: scaleX(0);
  transition: transform 0.3s ease;
}

.kvo-link-nav:hover::after,
.kvo-link-nav.active::after {
  transform: scaleX(1);
}
