HTML 
<label class="checkbox-modern">
  <input type="checkbox" />
  <span class="checkmark"></span>

</label>
CSS 
.checkbox-modern {
  display: inline-flex;
  align-items: center;
  gap: 12px;
  cursor: pointer;
}

.checkbox-modern input {
  display: none;
}

.checkbox-modern .checkmark {
  width: 20px;
  height: 20px;
  border: 2px solid rgba(255,255,255,0.3);
  border-radius: 6px;
  transition: all 0.3s;
}

.checkbox-modern input:checked ~ .checkmark {
  background: #667eea;
  border-color: #667eea;
  transform: scale(1.1);
}
